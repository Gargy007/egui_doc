74 static void D4D_BtnValue2Coor(D4D_OBJECT* pThis)
75 {
76  D4D_BUTTON* pBtn = D4D_GET_BUTTON(pThis);
77 
78  _calc.Position = D4D_GetClientToScreenPoint(pThis, &pThis->position);
79 
80  D4D_ComputeGeometry(&(_calc.contentGeom), pThis);
81 
82  //---------------- Get Max size of content -----------------
83  // 1. compare bitmap sizes
84  // Get normal bitmap size if exists
85 
86  _calc.ButSize = _calc.contentGeom.sz;
87 
88  if(pThis->initFlags & D4D_BTN_F_3D)
89  {
90  _calc.ButSize.cx -= D4D_BTN_3D_SHIFT;
91  _calc.ButSize.cy -= D4D_BTN_3D_SHIFT;
92  }
93 
94 // 3D Buttons Enabled
95  if(pThis->initFlags & D4D_BTN_F_3D)
96  {
97  _calc.ButPos = _calc.contentGeom.pnt;
98 
99  _calc.corGeom1.sz.cy = (D4D_COOR) D4D_BTN_3D_SHIFT;
100  _calc.corGeom2.sz.cx = (D4D_COOR) D4D_BTN_3D_SHIFT;
101 
102  _calc.corGeom1.pnt = _calc.corGeom2.pnt = _calc.ButPos;
103 
104 
105  #if (D4D_ROUND_CORNER_ENABLE != D4D_FALSE) //if Round corners
106 
107  _calc.ButRadius = 0;
108 
109  if(pThis->radius)
110  {
111  if(pThis->radius > D4D_BEVEL_WIDTH - 1)
112  _calc.ButRadius = (D4D_COOR)(pThis->radius - D4D_BEVEL_WIDTH);
113 
114  _calc.corGeom1.sz.cx = (D4D_COOR)(_calc.ButSize.cx - _calc.ButRadius * 2);
115  _calc.corGeom2.sz.cy = (D4D_COOR)(_calc.ButSize.cy - _calc.ButRadius * 2);
116 
117  _calc.corGeom1.pnt.x += _calc.ButRadius;
118  _calc.corGeom2.pnt.y += _calc.ButRadius;
119 
120 
121  }else
122  #endif
123  {
124  _calc.corGeom1.sz.cx = (D4D_COOR)(_calc.ButSize.cx + D4D_BTN_3D_SHIFT);
125  _calc.corGeom2.sz.cy = (D4D_COOR)(_calc.ButSize.cy + D4D_BTN_3D_SHIFT);
126  }
127 
128 
129  if(*(pBtn->pStatus) & D4D_BUTTON_STATUS_PRESSED_MASK)
130  {
131  _calc.ButPos.x += D4D_BTN_3D_SHIFT;
132  _calc.ButPos.y += D4D_BTN_3D_SHIFT;
133  }else
134  {
135  _calc.corGeom1.pnt.y += _calc.ButSize.cy;
136  _calc.corGeom2.pnt.x += _calc.ButSize.cx;
137 
138  }
139  }else
140  {
141  #if (D4D_ROUND_CORNER_ENABLE != D4D_FALSE) //if Round corners
142  _calc.ButRadius = pThis->radius;
143  if(_calc.ButRadius)
144  if(pThis->initFlags & D4D_OBJECT_F_FOCUSRECT)
145  _calc.ButRadius -= pThis->size.cx - _calc.contentGeom.sz.cx ;
146  #endif
147 
148  _calc.ButPos = _calc.contentGeom.pnt;
149  }
150  //D4D_BTN_3D_ENABLE
151 
152 }

161 static void D4D_BtnOnDraw(D4D_MESSAGE* pMsg)
162 {
163  D4D_OBJECT* pThis = pMsg->pObject;
164  D4D_BUTTON* pBtn = D4D_GET_BUTTON(pThis);
165  D4D_OBJECT_DRAWFLAGS draw = pMsg->prm.draw;
166  D4D_COLOR clrT, clrB;
167  const D4D_BMP* pBmp = pBtn->pBmpNormal;
168 
169 //-------- 3D Buttons Enabled --------
170  D4D_BMP_PROPERTIES bmpProp;
171 
172  bmpProp = (D4D_ALIGN_V_CENTER_MASK | D4D_ALIGN_H_CENTER_MASK);
173 
174  D4D_BtnValue2Coor(pThis);
175 
176  // focused bitmap (may be also NULL)
177 
178  clrT = D4D_ObjectGetForeColor(pThis, draw);
179  clrB = D4D_ObjectGetBckgColor(pThis, draw);
180 
181  if((draw & D4D_OBJECT_DRAWFLAGS_FOCUSED) && (pBtn->pBmpFocus))
182  pBmp = pBtn->pBmpFocus;
183 
184  if(pThis->initFlags & D4D_BTN_F_3D)
185  {
186  // Draw Button Corners
187  if(pThis->radius)
188  {
189  D4D_POINT shPos = _calc.Position;
190 
191  shPos.x += pThis->radius;
192  shPos.y += pThis->radius;
193  D4D_FillQuadrant(&shPos, pThis->radius, clrB, D4D_QUAD_4);
194  shPos.x += pThis->size.cx - pThis->radius * 2 - 1;
195  D4D_FillQuadrant(&shPos, pThis->radius, clrB, D4D_QUAD_1);
196  shPos.y += pThis->size.cy - pThis->radius * 2 - 1;
197  D4D_FillQuadrant(&shPos, pThis->radius, clrB, D4D_QUAD_2);
198  shPos.x -= (pThis->size.cx - pThis->radius * 2 - 1);
199  D4D_FillQuadrant(&shPos, pThis->radius, clrB, D4D_QUAD_3);
200  }
201 
202  // draw the shading
203  if(*(pBtn->pStatus) & D4D_BUTTON_STATUS_PRESSED_MASK)
204  {
205  if(pThis->pData->flags & D4D_OBJECT_F_BEVEL_MASK)
206  if((pThis->pData->flags & D4D_OBJECT_F_BEVEL_MASK) == D4D_OBJECT_F_BEVEL_RAISED)
207  {
208  pThis->pData->flags &= ~D4D_OBJECT_F_BEVEL_MASK;
209  pThis->pData->flags |= D4D_OBJECT_F_BEVEL_DENTED;
210  }else
211  {
212  pThis->pData->flags &= ~D4D_OBJECT_F_BEVEL_MASK;
213  pThis->pData->flags |= D4D_OBJECT_F_BEVEL_RAISED;
214  }
215 
216 
217 
218  D4D_DrawFrame(pThis, clrT, clrB);
219 
220  if(pThis->pData->flags & D4D_OBJECT_F_BEVEL_MASK)
221  if((pThis->pData->flags & D4D_OBJECT_F_BEVEL_MASK) == D4D_OBJECT_F_BEVEL_RAISED)
222  {
223  pThis->pData->flags &= ~D4D_OBJECT_F_BEVEL_MASK;
224  pThis->pData->flags |= D4D_OBJECT_F_BEVEL_DENTED;
225  }else
226  {
227  pThis->pData->flags &= ~D4D_OBJECT_F_BEVEL_MASK;
228  pThis->pData->flags |= D4D_OBJECT_F_BEVEL_RAISED;
229  }
230  }else
231  {
232  D4D_DrawFrame(pThis, clrT, clrB);
233  }
234 
235 
236 
237 
238 
239  D4D_FillRect(&_calc.corGeom1.pnt, &_calc.corGeom1.sz,clrB);
240  D4D_FillRect(&_calc.corGeom2.pnt, &_calc.corGeom2.sz,clrB);
241 
242  }else if(draw & (D4D_OBJECT_DRAWFLAGS_COMPLETE | D4D_OBJECT_DRAWFLAGS_STATE))
243  D4D_DrawFrame(pThis, clrT, clrB);
244 
245  // Draw the content of button
246 
247  pBtn->textBuff.str_properties->font_properties &= ~D4D_FNT_PRTY_TRANSPARENT_MASK;
248 
249  // Draw the bitmap
250  if(pBmp != NULL)
251  {
252  D4D_DrawRBmpRect(&_calc.ButPos,&_calc.ButSize, pBmp, bmpProp, clrB, (D4D_BOOL)!D4D_IsEnabled(pThis), pThis->radius);
253  pBtn->textBuff.str_properties->font_properties |= D4D_FNT_PRTY_TRANSPARENT_MASK;
254  }
255 
256  // draw the text
257  if(pBtn->textBuff.pText)
258  D4D_DrawTextRRectTab(&_calc.ButPos, &_calc.ButSize,&pBtn->textBuff, NULL, clrT, clrB, _calc.ButRadius);
259 
260  if(!pBtn->textBuff.pText && !pBmp)
261  D4D_FillRRect(&_calc.ButPos, &_calc.ButSize, clrB, pThis->radius);
262 
263  if(pThis->initFlags & D4D_BTN_F_CONT_RECT)
264  D4D_RRect(&_calc.ButPos, &_calc.ButSize, D4D_LINE_THIN, clrT, _calc.ButRadius);
265 
266 }

399 void D4D_BtnOnMessage(D4D_MESSAGE* pMsg)
400 {
401 #if defined(D4D_LLD_TCH) || defined(D4D_LLD_MOUSE)
402  D4D_OBJECT* pThis = pMsg->pObject;
403  D4D_BUTTON* pBtn = D4D_GET_BUTTON(pThis);
404 #endif
405 
406  switch(pMsg->nMsgId)
407  {
408  case D4D_MSG_DRAW:
409  D4D_BtnOnDraw(pMsg);
410  break;
411 
412  case D4D_MSG_KEYDOWN:
413  D4D_BtnOnKeyDown(pMsg);
414  break;
415 
416  case D4D_MSG_KEYUP:
417  D4D_BtnOnKeyUp(pMsg);
418  break;
419 
420  case D4D_MSG_KILLFOCUS:
421  D4D_BtnKillFocus(pMsg);
422  break;
423 
424  #if defined(D4D_LLD_TCH) || defined(D4D_LLD_MOUSE)
425  case D4D_MSG_MOUSE_BTN_LEFT_DOWN:
426  *(pBtn->pStatus) |= D4D_BUTTON_STATUS_MOUSE_CLICK_MASK;
427 
428  case D4D_MSG_TOUCHED:
429  D4D_BtnTouched(pMsg);
430  break;
431 
432  case D4D_MSG_MOUSE_ENTER:
433  if((*(pBtn->pStatus) & D4D_BUTTON_STATUS_MOUSE_CLICK_MASK) == 0)
434  break;
435 
436  case D4D_MSG_TOUCH_AUTO:
437  if((*(pBtn->pStatus) & (D4D_BUTTON_STATUS_PRESSED_MASK | D4D_BUTTON_STATUS_1ST_TOUCH_MASK)) == 0)
438  {
439  *(pBtn->pStatus) |= D4D_BUTTON_STATUS_PRESSED_MASK;
440  D4D_InvalidateObject(pThis, D4D_FALSE);
441  }
442  break;
443 
444  case D4D_MSG_MOUSE_LEAVE:
445  if((*(pBtn->pStatus) & D4D_BUTTON_STATUS_MOUSE_CLICK_MASK) == 0)
446  break;
447 
448  case D4D_MSG_TOUCH_LOST:
449  if((*(pBtn->pStatus) & (D4D_BUTTON_STATUS_PRESSED_MASK | D4D_BUTTON_STATUS_1ST_TOUCH_MASK)) == D4D_BUTTON_STATUS_PRESSED_MASK)
450  {
451  *(pBtn->pStatus) &= ~D4D_BUTTON_STATUS_PRESSED_MASK;
452  D4D_InvalidateObject(pThis, D4D_FALSE);
453  }
454  break;
455 
456  case D4D_MSG_MOUSE_BTN_LEFT_UP:
457  *(pBtn->pStatus) &= ~D4D_BUTTON_STATUS_MOUSE_CLICK_MASK;
458  case D4D_MSG_UNTOUCHED:
459  D4D_BtnUnTouched(pMsg);
460  break;
461  #endif
462  default:
463  // call the default behavior
464  D4D_ObjOnMessage(pMsg);
465  }
466 }
