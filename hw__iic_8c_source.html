140  // FUNCTION: HwIIC_SetBaudRate
141  // SCOPE: HW I2C peripheral low level driver function - Master mode
142  // DESCRIPTION: The function that try to compute and set the I2C frequency
143  //
144  // PARAMETERS: baudrate - final baudrate frequency
145  //
146  // RETURNS: differencies of real frequency to requested
147  //
148  //-----------------------------------------------------------------------------
149  sLWord HwIIC_SetBaudRate(LWord baudRate)
150  {
151  const Word scl_divider[] = {
152  20 , 22 , 24 , 26 , 28 , 30 , 34 , 40 , 28 , 32 , 36 , 40 , 44 ,
153  48 , 56 , 68 , 48 , 56 , 64 , 72 , 80 , 88 , 104, 128, 80 , 96 ,
154  112, 128, 144, 160, 192, 240, 160, 192, 224, 256, 288, 320, 384,
155  480, 320, 384, 448, 512, 576, 640, 768, 960, 640, 768, 896, 1024,
156  1152, 1280, 1536, 1920, 1280, 1536, 1792, 2048, 2304, 2560, 3072,
157  3840
158  };
159 
160  sLWord bestCaseVal = 0x80000000;
161  sLWord tmpVal;
162  Byte mul, icr;
163 
164 
165  for(mul = 0; mul < 3; mul++)
166  for(icr = 0; icr < 0x40; icr++)
167  {
168  tmpVal = (baudRate - (D4D_I2C_FREQ_BUS / ((1 << mul) * scl_divider[icr])));
169 
170  if(HwIIC_Abs(tmpVal) < HwIIC_Abs(bestCaseVal))
171  {
172  bestCaseVal = tmpVal;
173  i2c_f = ((mul)?(0x40 << mul) : 0) | icr;
174  }
175  }
176  HwIIC_DeInit();
177  HwIIC_Init();
178 
179  return bestCaseVal;
180  }

856  // SCOPE: HW I2C peripheral driver private function
857  // DESCRIPTION: The function send one byte on bus
858  //
859  // PARAMETERS: data - data to send
860  //
861  // RETURNS: result of write operation
862  // D4D_D4D_I2C_ - byte successfuly sent and receive ACK

871  // write data to I2C data register
872  D4D_I2C_D = data;
873 
874  // wait to finished send sequence
875  while (((D4D_I2C_S & D4D_I2C_S_I2CIF_MASK) == 0) && (++tmp)){};
876

878  if(tmp)
879  {
880  // Clear interrupt flag
881  D4D_I2C_S |= D4D_I2C_S_I2CIF_MASK;
882

891

893  return D4D_I2C_OK;
894  }

898 
899  // And notify
900  return D4D_I2C_FALSE;

906  // DESCRIPTION: The function read one byte on bus
907  //

909  // ack - acknowledge after read byte (for last reading should be NACK)
910  //

925

932  if(ack)

937

940 
941  return D4D_I2C_OK;
942  }
943

945  HwIIC_RestoreBus();
946
