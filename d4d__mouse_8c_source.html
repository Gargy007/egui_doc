143 {
144  LWord i;
145  void* d4d_MouseBmps[D4D_MOUSE_CURSOR_TYPES];
146 
147  if(pMouseCrs == NULL)
148  pMouseCursors = (D4D_MOUSECURSOR_BMP*)&d4d_mouseCursorBmp_default;
149  else
150  pMouseCursors = pMouseCrs;
151 
152 
153  for(i=0;i<D4D_MOUSE_CURSOR_TYPES;i++)
154  {
155  void* pCursorData = NULL;
156 
157  // check if the bitmap is OK
158  if(pMouseCursors->cursorArray[i])
159  {
160  // Check the right format
161  if(D4D_GetImgSrcType(pMouseCursors->cursorArray[i]) == D4D_EXTIMG_INTD4D)
162  {
163  D4D_SIZE size = D4D_GetBmpSize(pMouseCursors->cursorArray[i]);
164  D4DMOUSE_DESC* pCurDesc = D4D_LLD_MOUSE.D4DMOUSE_GetDescriptor();
165  LWord c_mode;
166 
167  if(D4D_GET_IMG_HEADER_2_FROM_FILENAME(pMouseCursors->cursorArray[i])->key == D4D_IMG_HEADER_2_KEY_D4DINT)
168  c_mode = D4D_GET_IMG_HEADER_2_FROM_FILENAME(pMouseCursors->cursorArray[i])->colorMode;
169  else
170  c_mode = D4D_GET_IMG_HEADER_FROM_FILENAME(pMouseCursors->cursorArray[i])->colorMode;
171 
172  //check the format of bitmap
173  if((size.cx == pCurDesc->size_x) && (size.cy == pCurDesc->size_y) && (c_mode == pCurDesc->bmpFormat)) // this is just only one format must be updated
174  {
175  if(D4D_GET_IMG_HEADER_2_FROM_FILENAME(pMouseCursors->cursorArray[i])->key == D4D_IMG_HEADER_2_KEY_D4DINT)
176  pCursorData = (void*)D4D_GET_IMG_HEADER_2_FROM_FILENAME(pMouseCursors->cursorArray[i])->pData;
177  else

179  }
180  }
181  }
182 
183 // if(i && (pCursorData == NULL))
184 // d4d_MouseBmps[i] = d4d_MouseBmps[0];
185 // else
186  d4d_MouseBmps[i] = pCursorData;
187  }
188 
189  D4D_LLD_MOUSE.D4DMOUSE_SetPointerBmp(d4d_MouseBmps, D4D_MOUSE_CURSOR_TYPES, pMouseCursors->keyColor);

353 
354  if(cursorType != D4D_MOUSE_CURSOR_TYPE_NORMAL)
355  return cursorType;
356

358  {
359  D4D_OBJECT* pParent = D4D_GetParentObject((D4D_OBJECT*)pObj);
360  if(pParent)
361  return D4D_ObjectGetMouseCursor(pParent);
362  else
363  D4D_ScreenGetMouseCursor(pObj->pData->pScreen);
364  }
365 
366  // return the type of mouse cursor

487 void D4D_HandleMouse(void)
488 {
489  // get active screen
490  D4D_SCREEN* pScreen = D4D_GetActiveScreen();
491  D4D_OBJECT** pObjects;
492  D4D_MOUSE mouseEvent;
493 
494  // Check the buffer of input keys changes
495  if(d4d_MouseBuff.writePos == d4d_MouseBuff.readPos)
496  return;
497 
498  // Read the latest record in the buffer of input key event
499  mouseEvent = d4d_MouseBuff.buff[d4d_MouseBuff.readPos++];
500 
501  // Check the overflow of the read pointer of readed input keys
502  if(d4d_MouseBuff.readPos >= D4D_MOUSE_BUFF_LENGTH)
503  d4d_MouseBuff.readPos = 0;
504 
505 
506  // prepare message
507  d4d_msg.pScreen = pScreen;
508  d4d_msg.prm.draw = 0;
509  d4d_msg.nMsgId = D4D_MSG_NONE;
510  switch(mouseEvent.event)
511  {
512  case D4D_MOUSE_EVENT_MOVE:
513  mouseEvent.x += d4d_mousePos.x;
514  mouseEvent.y += d4d_mousePos.y;
515 
516 
517  case D4D_MOUSE_EVENT_SETPOS:
518  // Correct the input coordination to physical screen
519  if(mouseEvent.x < 0)
520  mouseEvent.x = 0;
521  else if(mouseEvent.x >= (D4D_DisplayWidth() - 16))
522  mouseEvent.x = D4D_DisplayWidth() - 1 - 16;
523 
524  if(mouseEvent.y < 0)
525  mouseEvent.y = 0;
526  else if(mouseEvent.y >= (D4D_DisplayHeight() - 16))
527  mouseEvent.y = D4D_DisplayHeight() - 1 - 16;
528 
529  d4d_mousePos.x = mouseEvent.x;
530  d4d_mousePos.y = mouseEvent.y;
531 
532  // Manage the hover mouse above objects
533  if(d4d_MouseHoverObj)
534  {
535  D4D_BOOL validCoor;
536  // Check the coordination if its valid
537  if(d4d_MouseHoverObj->pObjFunc->CheckCoordinates)
538  validCoor = d4d_MouseHoverObj->pObjFunc->CheckCoordinates(d4d_MouseHoverObj, D4D_GetScreenToClientPoint(d4d_MouseHoverObj, &d4d_mousePos));
539  else
540  validCoor = D4D_ObjectCheckCoordinates(d4d_MouseHoverObj, D4D_GetScreenToClientPoint(d4d_MouseHoverObj, &d4d_mousePos));
541 
542  if(validCoor && D4D_HasObjectChildren(d4d_MouseHoverObj))
543  {
544  pObjects = (D4D_OBJECT**)&d4d_MouseHoverObj->pRelations[D4D_OBJECT_USR_DATA_CHILD_IX];
545 
546  while(*pObjects)
547  {
548  if(D4D_FindMouseHoverObject(*pObjects, d4d_mousePos))
549  {
550  validCoor = D4D_FALSE;
551  break;
552  }
553  pObjects++;
554  }
555  }
556 
557  if(!validCoor)
558  {
559  d4d_msg.pObject = d4d_MouseHoverObj;
560  d4d_MouseHoverObj = NULL;
561  d4d_msg.nMsgId = D4D_MSG_MOUSE_LEAVE;
562  D4D_SendMessage(&d4d_msg);
563  d4d_msg.nMsgId = D4D_MSG_NONE;
564  // Change the cursor bitmap when the cursor leaves the object
565  D4D_SetMouseCursorType((D4D_MOUSE_CURSOR_TYPE)((pScreen->pData->flags & D4D_SCR_FINT_MOUSE_MASK) >> D4D_SCR_FINT_MOUSE_SHIFT), D4D_FALSE);
566  }
567  }
568 
569  if(!d4d_MouseHoverObj)
570  {
571  pObjects = (D4D_OBJECT**)pScreen->pObjects;
572  D4D_OBJECT* pHoverObj;
573  // check all objects
574  while(*pObjects)
575  {
576  pHoverObj = D4D_FindMouseHoverObject(*pObjects, d4d_mousePos);
577  if(pHoverObj)
578  {
579  d4d_MouseHoverObj = pHoverObj;
580  break;
581  }
582  pObjects++;
583  }
584 
585  if(d4d_MouseHoverObj)
586  {
587  d4d_msg.nMsgId = D4D_MSG_MOUSE_ENTER;
588 
589  // change cursor D4D_OBJECT_F_MOUSE_MASK
590  D4D_SetMouseCursorType(D4D_ObjectGetMouseCursor(d4d_MouseHoverObj), D4D_FALSE);
591  }
592  }
593 
594  D4D_LLD_MOUSE.D4DMOUSE_SetCoor(d4d_mousePos.x, d4d_mousePos.y);
595  break;
596 
597  case D4D_MOUSE_EVENT_DOWN_L:
598  d4d_MouseStatus |= D4D_MOUSE_BTN_LEFT_MASK;
599 
600  if(d4d_MouseHoverObj)
601  {
602  if(d4d_LastClickedObj)
603  {
604  d4d_msg.pObject = d4d_LastClickedObj;
605  d4d_msg.nMsgId = D4D_MSG_MOUSE_BTN_LEFT_UP;
606  D4D_SendMessage(&d4d_msg);
607  }
608  }
609 
610  d4d_LastClickedObj = d4d_MouseHoverObj;
611 
612  d4d_msg.nMsgId = D4D_MSG_MOUSE_BTN_LEFT_DOWN;
613  break;
614 
615  case D4D_MOUSE_EVENT_UP_L:
616  d4d_MouseStatus &= ~D4D_MOUSE_BTN_LEFT_MASK;
617 
618  if(d4d_LastClickedObj)
619  {
620  if(d4d_LastClickedObj != d4d_MouseHoverObj)
621  {
622  d4d_msg.pObject = d4d_LastClickedObj;
623  d4d_msg.nMsgId = D4D_MSG_MOUSE_BTN_LEFT_UP;
624  D4D_SendMessage(&d4d_msg);
625  }
626  d4d_LastClickedObj = NULL;
627  }
628 
629  d4d_msg.nMsgId = D4D_MSG_MOUSE_BTN_LEFT_UP;
630  break;
631 
632  case D4D_MOUSE_EVENT_DOWN_R:
633  d4d_MouseStatus |= D4D_MOUSE_BTN_RIGHT_MASK;
634  d4d_msg.nMsgId = D4D_MSG_MOUSE_BTN_RIGHT_DOWN;
635  break;
636 
637  case D4D_MOUSE_EVENT_UP_R:
638  d4d_MouseStatus &= ~D4D_MOUSE_BTN_RIGHT_MASK;
639  d4d_msg.nMsgId = D4D_MSG_MOUSE_BTN_RIGHT_UP;
640  break;
641 
642  case D4D_MOUSE_EVENT_DOWN_M:
643  d4d_MouseStatus |= D4D_MOUSE_BTN_MIDDLE_MASK;
644  d4d_msg.nMsgId = D4D_MSG_MOUSE_BTN_MIDDLE_DOWN;
645  break;
646 
647  case D4D_MOUSE_EVENT_UP_M:
648  d4d_MouseStatus &= ~D4D_MOUSE_BTN_MIDDLE_MASK;
649  d4d_msg.nMsgId = D4D_MSG_MOUSE_BTN_MIDDLE_UP;
650  break;
651 
652  case D4D_MOUSE_EVENT_WHEEL_UP:
653  d4d_msg.nMsgId = D4D_MSG_MOUSE_BTN_WHEEL_UP;
654  break;
655 
656  case D4D_MOUSE_EVENT_WHEEL_DOWN:
657  d4d_msg.nMsgId = D4D_MSG_MOUSE_BTN_WHEEL_DOWN;
658  break;
659  }
660 
661  if((d4d_msg.nMsgId == D4D_MSG_MOUSE_ENTER) || (d4d_msg.nMsgId == D4D_MSG_MOUSE_LEAVE) || ((d4d_msg.nMsgId != D4D_MSG_NONE) && (((d4d_MouseStatus & D4D_MOUSE_CURSOR_MASK) >> D4D_MOUSE_CURSOR_SHIFT) == D4D_MOUSE_CURSOR_TYPE_NORMAL))) // any valid message to send
662  {
663 
664  // Handle Screen events
665  if(pScreen->pData->flags & D4D_SCR_FINT_MOUSE_MASK)
666  {
667  if(pScreen->flags & D4D_SCR_F_EXIT)
668  {
669  if(D4D_ScrCheckExitBtnCoor(pScreen, &d4d_mousePos))
670  {
671  D4D_EscapeScreen();
672  d4d_MouseHoverObj = NULL;
673  d4d_LastClickedObj = NULL;
674  return;
675  }
676  }
677 
678  if(D4D_ScrCheckCoor(pScreen, &d4d_mousePos))
679  {
680  d4d_msg.pObject = NULL;
681  // send the MOUSE EVENT message to screen
682  D4D_SendMessage(&d4d_msg);
683  }
684  }
685 
686  if(d4d_MouseHoverObj)
687  {
688  d4d_msg.pObject = d4d_MouseHoverObj;
