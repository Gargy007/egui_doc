115  // FUNCTION: D4DTCH_Init_CrTouch
116  // SCOPE: Low Level Driver API function
117  // DESCRIPTION: The function is used for initialization of this low level driver
118  //
119  // PARAMETERS: none
120  //
121  // RETURNS: result: 1 - Success
122  // 0 - Failed
123  //-----------------------------------------------------------------------------
124  static unsigned char D4DTCH_Init_CrTouch(void)
125  {
126  Byte status2;
127 
128  if(D4D_LLD_TCH_HW_CRTOUCH.D4DTCHHWCRTOUCH_Init == NULL)
129  return 0;
130 
131  if(D4D_LLD_TCH_HW_CRTOUCH.D4DTCHHWCRTOUCH_DeInit == NULL)
132  return 0;
133 
134  if(D4D_LLD_TCH_HW_CRTOUCH.D4DTCHHWCRTOUCH_MultiReadBytes == NULL)
135  return 0;
136 
137 
138  if(D4D_LLD_TCH_HW_CRTOUCH.D4DTCHHWCRTOUCH_Init() != D4D_LLD_TCH_HW_CRTOUCH.COMM_OK)
139  return 0;
140 
141  if(D4D_LLD_TCH_HW_CRTOUCH.D4DTCHHWCRTOUCH_SetBaudRate != NULL)
142  D4D_LLD_TCH_HW_CRTOUCH.D4DTCHHWCRTOUCH_SetBaudRate(D4DTCH_IIC_BAUDRATE);
143 
144  if(D4D_LLD_TCH_HW_CRTOUCH.D4DTCHHWCRTOUCH_SetAddress != NULL)
145  D4D_LLD_TCH_HW_CRTOUCH.D4DTCHHWCRTOUCH_SetAddress(D4DTCH_IIC_ADDRESS);
146 
147  if(D4D_LLD_TCH_HW_CRTOUCH.D4DTCHHWCRTOUCH_MultiReadBytes(CRTCH_STATUS2, &status2, 1) != D4D_LLD_TCH_HW_CRTOUCH.COMM_OK)
148  {
149  (void)D4D_LLD_TCH_HW_CRTOUCH.D4DTCHHWCRTOUCH_DeInit();
150  return 0;
151  }
152 
153  if((status2 & CRTCH_STATUS2_RTSD_MASK) == 0)
154  {
155  // The touch screen hardware is NOT detected
156  (void)D4D_LLD_TCH_HW_CRTOUCH.D4DTCHHWCRTOUCH_DeInit();
157  return 0;
158  }
159 
160  // Initializations is OK
161  return 1;
162  }

195  //-----------------------------------------------------------------------------
196  // FUNCTION: D4DTCH_GetPositionRaw_CrTouch
197  // SCOPE: Low Level Driver API function
198  // DESCRIPTION: Reads touch screen and returns raw uncompensated X, Y
199  // coordinates if screen touched
200  // PARAMETERS: unsigned short *TouchPositionX Pointer to X coordinate
201  // unsigned short *TouchPositionY Pointer to Y ccordinate
202  // RETURNS: 0 no screen touch
203  // 1 screen touch,
204  //-----------------------------------------------------------------------------
205  static unsigned char D4DTCH_GetPositionRaw_CrTouch(unsigned short *TouchPositionX,
206  unsigned short *TouchPositionY)
207  {
208 
209  static unsigned char last_state = 0;
210  static unsigned short lastX, lastY;
211  CRTOUCH_RES_SIMPLE crtouch_res;
212 
213  if(D4D_LLD_TCH_HW_CRTOUCH.D4DTCHHWCRTOUCH_SetAddress != NULL)
214  D4D_LLD_TCH_HW_CRTOUCH.D4DTCHHWCRTOUCH_SetAddress(D4DTCH_IIC_ADDRESS);
215 
216  if(D4D_LLD_TCH_HW_CRTOUCH.D4DTCHHWCRTOUCH_MultiReadBytes( CRTCH_STATUS1, (Byte*)(&crtouch_res), sizeof(crtouch_res)) == D4D_LLD_TCH_HW_CRTOUCH.COMM_OK)
217  {
218  if(crtouch_res.status1 & CRTCH_STATUS1_RTSRDY_MASK)
219  {
220  if(crtouch_res.status1 & CRTCH_STATUS1_RTST_MASK)
221  {
222  #if D4DTCH_SWAP_AXES == 0
223  lastX = D4D_READ16B(crtouch_res.X);
224  lastY = D4D_READ16B(crtouch_res.Y);
225  #else
226  lastX = D4D_READ16B(crtouch_res.Y);
227  lastY = D4D_READ16B(crtouch_res.X);
228  #endif
229 
230  #ifdef D4DTCH_INVERSE_X_INT
231  lastX = D4DTCH_INVERSE_X_INT - lastX;
232  #endif
233 
234  #ifdef D4DTCH_INVERSE_Y_INT
235  lastY = D4DTCH_INVERSE_Y_INT - lastY;
236  #endif
237 
238  last_state = 1;
239  }
240  else
241  last_state = 0;
242  }
243  }
244 
245  *TouchPositionX = lastX;
246  *TouchPositionY = lastY;
247  return last_state;
248  }
