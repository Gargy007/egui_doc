212  // Registrer interrupt service routines

223

258 #if D4DLCDHWFB_DMA_ENABLE
259  SIM_SCGC7 |= SIM_SCGC7_DMA_MASK;

261  #if D4DLCDHWFB_DMA_CHANNEL < 16
262  SIM_SCGC6 |= SIM_SCGC6_DMAMUX0_MASK;

269 
270  // set up the channel priority
271  *(unsigned char*)((unsigned long)&DMA_DCHPRI3 + (3 - (D4DLCDHWFB_DMA_CHANNEL % 4)) + ((D4DLCDHWFB_DMA_CHANNEL / 4) * 4)) = D4DLCDHWFB_DMA_CHPRI;

289  _bsp_int_init(D4DLCDHWFB_DMA_INTVECT, 3, 0, TRUE);
290  #else
291  NVIC_IP(D4DLCDHWFB_DMA_INTVECT) = 3;
292  NVIC_ISER(D4DLCDHWFB_DMA_INTVECT / 32) |= (1 << (D4DLCDHWFB_DMA_INTVECT % 32));
293  #warning The LCDC K70 driver is is using interrupts, please registry the "D4DLCDHW_ISR_K70DMA" on right interrupt vector for selected DMA channel by D4DLCDHWFB_DMA_CHANNEL.
294  #endif
295 
296 #endif
297  // Clear the previous contents of buffer
298  pTmp = (unsigned long*)fb_start_addr_2nd;
299  for(i=0;i<(D4DLCDHWFB_X_MAX * D4DLCDHWFB_Y_MAX);i++)
300  *(pTmp++) = 0; // Black color
301 
302  #endif
303 
304  enableWrite = 1;
305 #ifdef LCDC_DEBUG
306  GPIOB_PSOR |= 1<<10;
307 #endif
308 
309  #if D4DLCDHWFB_MIRROWED == 1
310  LCDC_LSSAR = d4dlcdhw_k70_lcdc_desc.fb_start_addr + D4DLCDHWFB_X_MAX * D4DLCDHWFB_Y_MAX * D4DLCDHWFB_BPP_BYTE;
311  #else
312  LCDC_LSSAR = d4dlcdhw_k70_lcdc_desc.fb_start_addr;
313  #endif
314 
315 #ifdef LCDC_DEBUG
316  GPIOB_PSOR |= 1<<11;
317 #endif
318 
319  LCDC_LSR = (D4DLCDHWFB_X_MAX / 16) << 20 | (D4DLCDHWFB_Y_MAX);
320

324  // set LCD cursor positon & settings (turn off)

326  LCDC_LCWHB = 0;

340  D4D_LLD_LCD_HW_K70LCDC.D4DLCDHWFBK70LCDC_InitPanel();
341

344  #endif
345  /* Enable LCD */

348 
349  return 1;

411 
412

423  {

427 
428  #endif
