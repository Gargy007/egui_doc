250 int D4D_CompareStrings(const D4D_CHAR* s1, const D4D_CHAR* s2)
251 {
252 #ifndef D4D_STD_STRING_H_INCLUDE
253  D4D_CHAR val2 = *s2++;
254  while(val2)
255  {
256  if (*s1++ != val2)
257  {
258  if (s1[-1] < val2)
259  return -1;
260 
261  return 1;
262  }
263  val2 = *s2++;
264  }
265 
266  if (*s1) // if we get to here, then s2 terminated
267  return 1; // s1 is still going
268 
269  return 0; // they both terminated
270 #else
271  // Standard system implementation
272  return strcmp(s1, s2);
273 #endif
274 }
