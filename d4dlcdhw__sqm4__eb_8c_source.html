118 static unsigned char D4DLCDHW_Init_K70LCDC_Sqm4Eb(void)
119 {
120  /* Setup LCD pin muxing */
121  lcdc_init_pins();
122 
123  // set LCD panning offset
124  LCDC_LPOR = 0;
125 
126  // set LCD panel configuration
127  LCDC_LPCR =
128  LCDC_LPCR_TFT_MASK | //TFT Screen
129  LCDC_LPCR_COLOR_MASK | //Color
130  LCDC_LPCR_BPIX(D4DLCDHWFB_BPP) | //bpp
131  //LCDC_LPCR_PIXPOL_MASK | // pixel polarity
132  LCDC_LPCR_FLMPOL_MASK | //first line marker active low
133  LCDC_LPCR_LPPOL_MASK | //line pulse active low
134  LCDC_LPCR_END_SEL_MASK | //Use big-endian mode (0xFFAA5500 means R=AA,G=55,B=00)
135  //LCDC_LPCR_SWAP_SEL_MASK |
136  LCDC_LPCR_SCLKIDLE_MASK | //Enalbe LSCLK when vsync is idle
137  LCDC_LPCR_SCLKSEL_MASK | //Always enable clock
138  LCDC_LPCR_CLKPOL_MASK | //Active on positive edge of LSCLK. In TFT mode, active on negative edge of LSCLK.
139  // LCDC_LPCR_ACD(ACD_DIV_7) | // It isn't used in TFT mode
140  LCDC_LPCR_PCD(D4DLCDHWFB_PANEL_CLKDIV); //Divide 120 PLL clock by (12+1)=13 to get 9.23MHz clock
141 
142  // set LCD horizontal configuration based on panel data (Figure 3-3 in Seiko datasheet)
143 
144 #if defined(FRD7040TPT) // -----------------------------------------------
145  LCDC_LHCR =
146  LCDC_LHCR_H_WIDTH(5) | //(47+1)=48 SCLK period for HSYNC activated
147  LCDC_LHCR_H_WAIT_1(0x20) | //(39+1)=40 SCLK period between end of OE and beginning of HSYNC
148  LCDC_LHCR_H_WAIT_2(0x10); //(85+3)=88 SCLK periods between end of HSYNC and beginning of OE
149 
150  // set LCD vertical configuration based on panel data (Figure 3-3 in Seiko datasheet)
151  LCDC_LVCR =
152  LCDC_LVCR_V_WIDTH(0x30) | //3 lines period for VSYNC activated
153  LCDC_LVCR_V_WAIT_1(0x30) | //13 line period between end of OE and beginning of VSYNC
154  LCDC_LVCR_V_WAIT_2(0x30); //32 line periods between end of VSYNC and beginning of OE
155 
156 #elif defined(FRD5040TPT) // -----------------------------------------------
157  LCDC_LHCR =
158  LCDC_LHCR_H_WIDTH(47) | //(47+1)=48 SCLK period for HSYNC activated
159  LCDC_LHCR_H_WAIT_1(39) | //(39+1)=40 SCLK period between end of OE and beginning of HSYNC
160  LCDC_LHCR_H_WAIT_2(85); //(85+3)=88 SCLK periods between end of HSYNC and beginning of OE
161 
162  // set LCD vertical configuration based on panel data (Figure 3-3 in Seiko datasheet)
163  LCDC_LVCR =
164  LCDC_LVCR_V_WIDTH(3) | //3 lines period for VSYNC activated
165  LCDC_LVCR_V_WAIT_1(13) | //13 line period between end of OE and beginning of VSYNC

167 
168 #else // -----------------------------------------------
169  LCDC_LHCR =
170  LCDC_LHCR_H_WIDTH(40) | //(40+1)=41 SCLK period for HSYNC activated

179 
180 #endif // -----------------------------------------------
181
