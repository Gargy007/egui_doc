369 static void spi_init(void)
370 {
371  /* clock gate */
372  SIM_SCGC3 |=SIM_SCGC3_DSPI2_MASK ;
373 
374  //Turn on SPI2 Mux
375  PORTD_PCR11 = PORT_PCR_MUX(2); //SPI2_PCS0
376  PORTD_PCR12 = PORT_PCR_MUX(2); //SPI2_SCK
377  PORTD_PCR13 = PORT_PCR_MUX(2); //SPI2_SOUT
378  PORTD_PCR14 = PORT_PCR_MUX(2); //SPI2_SIN
379 
380  SPI2_MCR &= (~ SPI_MCR_MDIS_MASK);
381  SPI2_MCR = SPI_MCR_HALT_MASK | SPI_MCR_CLR_TXF_MASK | SPI_MCR_CLR_RXF_MASK | SPI_MCR_DIS_RXF_MASK;
382  SPI2_CTAR0 = SPI_CTAR_FMSZ(15) | SPI_CTAR_BR(2);
383  SPI2_CTAR1 = SPI_CTAR_FMSZ(15) | SPI_CTAR_BR(2);
384 
385 
386  // Receive FIFO overflow disable
387  SPI2_MCR |= SPI_MCR_ROOE_MASK;
388 
389  // Set CS0-7 inactive high
390  SPI2_MCR |= SPI_MCR_PCSIS(0xFF);
391  SPI2_MCR |= SPI_MCR_MSTR_MASK;
392 
393  // Disable interrupts
394  SPI2_RSER = 0;
395 
396  // Clear all flags
397  SPI2_SR |= SPI_SR_TFFF_MASK | SPI_SR_TCF_MASK;
398 
399  // Enable SPI
400  SPI2_MCR &= (~ SPI_MCR_HALT_MASK);
401 }

403 /*
404  * To set the LCD register to read/write, send out 0x70 and then the register #
405  */
406 static void SpiRegSet(unsigned char reg)
407 {
408  unsigned short data;
409 
410  //Set the data to write out by concatenating 0x70 with the register value.
411  data=0x7000 | reg;
412 
413  // wait write buffer not full flag
414  while (!(SPI2_SR & SPI_SR_TFFF_MASK)){};
415 
416  // Assert CS0, Use config 0, and send out on MISO
417  SPI2_PUSHR = SPI_PUSHR_PCS(1 << (0)) | SPI_PUSHR_CTAS(0) | SPI_PUSHR_TXDATA((unsigned short)data);
418 
419  while (!(SPI2_SR & SPI_SR_TCF_MASK)){};// while shift-out complete
420  SPI2_SR = SPI_SR_TCF_MASK; // clear flag
421 }

446 /*
447  * To read data from a LCD register, this function sets the register index,
448  * and then sends out 0x73 and wait for reply
449  */
450 static unsigned short SpiReadDataWord(unsigned char reg)
451 {
452  unsigned short value, data;
453  unsigned short data2;
454 
455  //Set register pointer
456  data=0x7000 | reg;
457 
458  // wait write buffer not full flag
459  while (!(SPI2_SR & SPI_SR_TFFF_MASK)){};
460 
461  // Assert CS0, Use config 0
462  SPI2_PUSHR = SPI_PUSHR_PCS(1 << (0)) | SPI_PUSHR_CTAS(0) | SPI_PUSHR_TXDATA((unsigned short)data);
463 
464  while (!(SPI2_SR & SPI_SR_TCF_MASK)){};// while shift-out complete
465  SPI2_SR = SPI_SR_TCF_MASK; // clear flag
466 
467 
468  //Tell it to do a read by sending out 0x73
469  data=0x7300;
470  data2=0x0000;
471 
472  //Halt SPI from sending out anything
473  SPI2_MCR |= SPI_MCR_HALT_MASK;
474 
475  // wait write buffer not full flag
476  while (!(SPI2_SR & SPI_SR_TFFF_MASK)){};
477 
478  // Assert CS0, Use config 0, write two words at once to have CW low long enough
479  // to let data be recieved.
480  SPI2_PUSHR = SPI_PUSHR_CONT_MASK | SPI_PUSHR_PCS(1 << (0)) | SPI_PUSHR_CTAS(0) | SPI_PUSHR_TXDATA((unsigned short)data);
481  SPI2_PUSHR = SPI_PUSHR_PCS(1 << (0)) | SPI_PUSHR_CTAS(0) | SPI_PUSHR_TXDATA((unsigned short)data2);
482 
483 
484  //Send out both data words back to back using the FIFO
485  SPI2_MCR &= (~ SPI_MCR_HALT_MASK);
486 
487  while (!(SPI2_SR & SPI_SR_TCF_MASK)){};// while shift-out complete
488  //while (!(SPI2_SR & SPI_SR_RCF_MASK)){};// while shift-out complete
489  SPI2_SR = SPI_SR_TCF_MASK; // clear flag
490 
491  time_delay_ms(5);
492 
493  //Data returned from LCD should be in POPR now
494  value = SPI2_POPR ; //garbage
495  value = SPI2_POPR ; //good data
496 
497  return value;
498 }
